#include <iostream>
#include <fstream>
using namespace std;
ifstream fin("grad.in");
ofstream fout("grad.out");
int D[50];
int n,m;
void citire()
{
   int x,y,i;
   fin>>n>>m;
   for(i=1;i<=m;i++)
   {
       fin>>x>>y;
       D[x]++;
       D[y]++;
   }
}
void afis()
{
    int i,j;
    for(i=1;i<=n;i++) 
    fout<<D[i]<<" ";
    fout<<endl;
    for(i=1;i<=n;i++) 
        if(D[i]%2==0) 
        fout<<i<<" ";
}
int main()
{
    citire();
    afis();
    fin.close();
    fout.close();
    return 0;
}
