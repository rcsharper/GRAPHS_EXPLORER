#include <iostream>
#include <fstream>
using namespace std;
ifstream fin("grad.in");
ofstream fout("grad.out");
int a[10][10],n,gradext;
void citire()
{
  int x,y;
  f>>n>>m;
  for(int i=1;i<=m;i++)
   {
    f>>x>>y;
    a[x][y]=1;
    a[y][x]=1;
   }
 }
int main()
{
    citire();
    for (i = 1; i <= n; i++)
    {
    grafext = 0;
    for (j = 1; j <= n; j++)
    gradext = gradext + a[i, j];
    }
    cout<<"(d +( " + i + " ) =" + gradext;               
    fin.close();
    fout.close();
    return 0;
}
